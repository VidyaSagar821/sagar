/// <summary>
/// Get a list with all packages by status and useremail.
/// </summary>
/// <param name="status" example="0">The package status(Draft(0), Pending(1), Approved(2), Rejected(3), AddToExisting(4))</param>
/// <param name="useremail" example="john_doe@worldvision.ca">The email of the user who created the package</param>
/// <response code="200">Returns a list with the available packages with the matching status.</response>
// GET: api/<PackageController>
[HttpGet("/GetPackagesByStatus")]
public async Task<List<Package>> GetPackages([FromQuery(Name = "status")] PackageStatus status, [FromQuery(Name = "useremail")] string userEmail)
{
    return await _packageService.GetPackages(Enum.GetName(status.GetType(), status), userEmail);
}
